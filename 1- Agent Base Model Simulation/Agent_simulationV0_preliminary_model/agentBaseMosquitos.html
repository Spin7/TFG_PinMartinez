import numpy as np
from scipy.integrate import solve_ivp

# Parámetros
b, k, d, mu1, mu2 = 4, 1000, 0.05, 1/30, 1/30
J0, A0 = 200, 200
t_span = [0, 100]
ti = np.linspace(t_span[0], t_span[1], 1000)

# Ecuaciones diferenciales
def deriv(t, y):
    J, A = y
    dJdt = b * (1 - J/k) * A - (d + mu1) * J
    dAdt = 0.5 * d * J - mu2 * A
    return [dJdt, dAdt]

# Solución determinista
ret = solve_ivp(deriv, t_span, [J0, A0], t_eval=ti)

A = ret.y[1]

import numpy as np
from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

# Parámetros del modelo
b, k, d, mu1, mu2 = 4, 1000, 0.05, 1/30, 1/30
J0, A0 = 200, 200
t_span = [0, 100]
ti = np.linspace(t_span[0], t_span[1], 1000)

# 1. Solución determinista (original)
def deriv(t, y):
    J, A = y
    dJdt = b * (1 - J/k) * A - (d + mu1) * J
    dAdt = 0.5 * d * J - mu2 * A
    return [dJdt, dAdt]

ret = solve_ivp(deriv, t_span, [J0, A0], t_eval=ti)
J_det = ret.y[0]
A_det = ret.y[1]

tii = np.linspace(t_span[0], t_span[1], 100)

# 2. Simulación estocástica corregida (Gillespie)
def stochastic_simulation(J0, A0, t_span, tii):
    current_time = t_span[0]
    current_J, current_A = J0, A0
    
    # Listas para almacenar el historial de tiempos y estados
    time_points = [current_time]
    J_values = [current_J]
    A_values = [current_A]
    
    while current_time < t_span[1]:
        # Calcular las tasas de transición
        lambda_J = b * (1 - current_J / k) * current_A
        mu_J = (d + mu1) * current_J
        lambda_A = 0.5 * d * current_J
        mu_A = mu2 * current_A
        
        rates = [lambda_J, mu_J, lambda_A, mu_A]
        total_rate = sum(rates)
        
        # Si no hay eventos posibles, terminar
        if total_rate <= 0:
            break
        
        # Generar tiempo hasta el próximo evento
        delta_time = np.random.exponential(1 / total_rate)
        current_time += delta_time
        
        # Si se excede el tiempo máximo, terminar
        if current_time > t_span[1]:
            break
        
        # Seleccionar el evento basado en las tasas
        event_probs = np.array(rates) / total_rate
        event = np.random.choice(4, p=event_probs)
        
        # Actualizar el estado según el evento
        if event == 0:    # Nacimiento de J
            current_J += 1
        elif event == 1:  # Muerte de J
            current_J -= 1
        elif event == 2:  # Maduración J -> A
            current_J -= 1
            current_A += 1
        elif event == 3:  # Muerte de A
            current_A -= 1
        
        # Asegurar que las poblaciones no sean negativas
        current_J = max(current_J, 0)
        current_A = max(current_A, 0)
        
        # Registrar el estado y el tiempo actual
        time_points.append(current_time)
        J_values.append(current_J)
        A_values.append(current_A)
    
    # Muestrear los estados en los tiempos de interés (ti)
    indices = np.searchsorted(time_points, tii, side='right') - 1
    indices = np.clip(indices, 0, len(J_values)-1)  # Evitar índices negativos
    
    J_stoch = np.array(J_values)[indices]
    A_stoch = np.array(A_values)[indices]
    
    return J_stoch, A_stoch

# Ejecutar simulación estocástica
J_stoch, A_stoch = stochastic_simulation(J0, A0, t_span, tii)

# Graficar resultados
plt.figure(figsize=(10, 6))
plt.plot(ti, J_det, label='J (determinista)', color='blue')
plt.plot(ti, A_det, label='A (determinista)', color='orange')
plt.plot(tii, J_stoch, '--', label='J (estocástico)', color='blue', alpha=0.7)

plt.xlabel('Tiempo')

plt.legend()
plt.title('Comparación entre modelos determinista y estocástico')

import heapq
import random
from itertools import count

class AgentBasedModel:
    def __init__(self, J0, A0, params):
        self.b, self.k, self.d, self.mu1, self.mu2 = params
        self.current_time = 0.0
        self.J_count = J0
        self.A_count = A0
        self.J_alive = {}  # Track alive J agents
        self.A_alive = {}  # Track alive A agents
        self.event_queue = []
        self.id_gen = count()  # Unique agent IDs

        # Initialize J agents
        for _ in range(J0):
            j_id = next(self.id_gen)
            self.J_alive[j_id] = True
            self._schedule_j_events(j_id)

        # Initialize A agents
        for _ in range(A0):
            a_id = next(self.id_gen)
            self.A_alive[a_id] = True
            self._schedule_a_events(a_id)

    def _schedule_j_events(self, j_id):
        # Schedule death and maturation for J
        death_time = self.current_time + random.expovariate(self.d + self.mu1)
        heapq.heappush(self.event_queue, (death_time, 'death_J', j_id))
        mature_time = self.current_time + random.expovariate(0.5 * self.d)
        heapq.heappush(self.event_queue, (mature_time, 'mature_J', j_id))

    def _schedule_a_events(self, a_id):
        # Schedule death and reproduction for A
        death_time = self.current_time + random.expovariate(self.mu2)
        heapq.heappush(self.event_queue, (death_time, 'death_A', a_id))
        repro_time = self.current_time + random.expovariate(self.b)
        heapq.heappush(self.event_queue, (repro_time, 'reproduce_A', a_id))

    def run(self, t_max):
        time_history = [0.0]
        J_history = [self.J_count]
        A_history = [self.A_count]

        while self.event_queue and self.current_time <= t_max:
            event_time, event_type, agent_id = heapq.heappop(self.event_queue)
            self.current_time = event_time

            if event_type == 'death_J':
                if agent_id in self.J_alive:
                    del self.J_alive[agent_id]
                    self.J_count -= 1
            elif event_type == 'mature_J':
                if agent_id in self.J_alive:
                    del self.J_alive[agent_id]
                    self.J_count -= 1
                    # Create new adult
                    a_id = next(self.id_gen)
                    self.A_alive[a_id] = True
                    self.A_count += 1
                    self._schedule_a_events(a_id)
            elif event_type == 'death_A':
                if agent_id in self.A_alive:
                    del self.A_alive[agent_id]
                    self.A_count -= 1
            elif event_type == 'reproduce_A':
                if agent_id in self.A_alive:
                    # Attempt reproduction with density-dependent success
                    prob = max(0, 1 - self.J_count / self.k)
                    if random.random() < prob:
                        j_id = next(self.id_gen)
                        self.J_alive[j_id] = True
                        self.J_count += 1
                        self._schedule_j_events(j_id)
                    # Schedule next reproduction
                    next_repro_time = self.current_time + random.expovariate(self.b)
                    heapq.heappush(self.event_queue, (next_repro_time, 'reproduce_A', agent_id))

            # Record state at event time
            time_history.append(self.current_time)
            J_history.append(self.J_count)
            A_history.append(self.A_count)

        return time_history, J_history, A_history

# Parameters
params = (4, 1000, 0.05, 1/30, 1/30)
model = AgentBasedModel(J0=200, A0=200, params=params)
time_points, J_counts, A_counts = model.run(t_max=100)

# Plot results

import random
import matplotlib.pyplot as plt

class AgentBasedModel:
    def __init__(self, J0, A0, params):
        self.b, self.k, self.d, self.mu1, self.mu2 = params
        self.current_time = 0.0
        self.J_count = J0
        self.A_count = A0
        self.J_alive = set(range(J0))  # Track alive J agents
        self.A_alive = set(range(J0, J0 + A0))  # Track alive A agents
        self.id_gen = count(J0 + A0)  # Unique agent IDs

    def run(self, t_max, delta_t):
        time_history = [0.0]
        J_history = [self.J_count]
        A_history = [self.A_count]

        while self.current_time < t_max:
            self.current_time += delta_t

            # Process J agents (aquatic mosquitoes)
            for j_id in list(self.J_alive):
                # Death event
                if random.random() < (self.d + self.mu1) * delta_t:
                    self.J_alive.remove(j_id)
                    self.J_count -= 1
                # Maturation event
                elif random.random() < (0.5 * self.d) * delta_t:
                    self.J_alive.remove(j_id)
                    self.J_count -= 1
                    # Create new adult
                    a_id = next(self.id_gen)
                    self.A_alive.add(a_id)
                    self.A_count += 1

            # Process A agents (adult mosquitoes)
            for a_id in list(self.A_alive):
                # Death event
                if random.random() < self.mu2 * delta_t:
                    self.A_alive.remove(a_id)
                    self.A_count -= 1
                # Reproduction event
                elif random.random() < self.b * delta_t:
                    # Density-dependent success
                    prob = max(0, 1 - self.J_count / self.k)
                    if random.random() < prob:
                        j_id = next(self.id_gen)
                        self.J_alive.add(j_id)
                        self.J_count += 1

            # Record state at current time
            time_history.append(self.current_time)
            J_history.append(self.J_count)
            A_history.append(self.A_count)

        return time_history, J_history, A_history

Este código implementa un modelo de agentes en el que cada mosquito es tratado como una entidad individual, y su evolución es gobernada por procesos de Bernoulli en pasos de tiempo discretos ΔtΔt. A diferencia del método de Gillespie, que usa tiempos de espera exponenciales entre eventos, aquí se actualizan todos los agentes en cada iteración con una probabilidad dada.

  1. Proceso de Markov subyacente

El modelo puede representarse como un proceso de Markov con estados (J,A)(J,A), donde:

JJ representa el número de mosquitos en estado acuático.
AA representa el número de mosquitos adultos.
La evolución del sistema depende de eventos con probabilidades dadas.

Las transiciones posibles entre estados son:

Muerte de J: J→J−1J→J−1 con probabilidad (d+μ1)Δt(d+μ1​)Δt.
Muerte de A: A→A−1A→A−1 con probabilidad μ2Δtμ2​Δt.
Maduración de J a A: J→J−1,A→A+1J→J−1,A→A+1 con probabilidad 0.5dΔt0.5dΔt.
Nacimiento de nuevos J: J→J+1J→J+1 con probabilidad bΔt⋅(1−J/k)bΔt⋅(1−J/k) si la población acuática no está saturada.
  1. Matriz de transición PP

La matriz de transición define la probabilidad de pasar de un estado (J,A)(J,A) a otro en un solo paso de tiempo. Para un sistema discreto, la matriz PP tiene la forma: P=[P(J,A)→(J+1,A)P(J,A)→(J−1,A)P(J,A)→(J,A+1)P(J,A)→(J,A−1)P(J,A)→(J−1,A+1)] P=[P(J,A)→(J+1,A)​​P(J,A)→(J−1,A)​​P(J,A)→(J,A+1)​​P(J,A)→(J,A−1)​​P(J,A)→(J−1,A+1)​​]

Para un estado (J,A)(J,A), los términos de transición son:

P(J,A)→(J+1,A)=bΔt⋅(1−J/k)P(J,A)→(J+1,A)​=bΔt⋅(1−J/k) (Nacimiento de J)
P(J,A)→(J−1,A)=(d+μ1)ΔtP(J,A)→(J−1,A)​=(d+μ1​)Δt (Muerte de J)
P(J,A)→(J,A+1)=0.5dΔtP(J,A)→(J,A+1)​=0.5dΔt (Maduración de J a A)
P(J,A)→(J,A−1)=μ2ΔtP(J,A)→(J,A−1)​=μ2​Δt (Muerte de A)
P(J,A)→(J−1,A+1)=0.5dΔtP(J,A)→(J−1,A+1)​=0.5dΔt (Maduración de J a A)

Condición de normalización

Como es un proceso de Markov, las probabilidades deben sumar 11: P(J,A)→(J,A)=1−[P(J,A)→(J+1,A)+P(J,A)→(J−1,A)+P(J,A)→(J,A+1)+P(J,A)→(J,A−1)+P(J,A)→(J−1,A+1)] P(J,A)→(J,A)​=1−[P(J,A)→(J+1,A)​+P(J,A)→(J−1,A)​+P(J,A)→(J,A+1)​+P(J,A)→(J,A−1)​+P(J,A)→(J−1,A+1)​]

Si la suma de las transiciones supera 1, se debe reducir ΔtΔt para que todas las probabilidades estén en el rango [0,1][0,1].
